  <ok href="/term/1045">
   美国
  </ok>

  <ok href="/term/1045">

  <ok href="/term/1045">

 </p>

  <ok href="/term/1045">
   美国
  </ok>
  所有造船厂的总和。

 </p>
 <p>

  <ok href="/term/1045">
   美国
  </ok>

  <ok href="/term/1045">
   美国
  </ok>

 </p>
 <p>

  <ok href="/term/1045">
   美国
  </ok>

  <ok href="/term/1045">
   美国
  </ok>

 </p>
 <p>

  <ok href="/term/1045">
